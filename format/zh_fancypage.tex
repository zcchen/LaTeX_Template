\def\emptyTest#1#2{
    % #1 = varialbe
    % #2 = format
    \ifx#1\empty% ignore it
    \else
        #2
        #1
        \vskip 1ex
    \fi
}
\def\maketitle{%
    \vskip 5em
    \begin{center}
        \fangsong%
        \emptyTest{\@title}{\Huge}
        \emptyTest{\@author}{\large}
        \emptyTest{\@date}{\large}
    \end{center}
    \global\let\maketitle\relax
    \global\let\@author\@empty%
    \global\let\@date\@empty%
    \global\let\@title\@empty%
    \global\let\title\relax
    \global\let\author\relax
    \global\let\date\relax
    \global\let\and\relax
}
